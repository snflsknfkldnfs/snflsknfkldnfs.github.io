        self.repo = git.Repo(repo_path)

            'M6': self.load_template('M6'),
            'E5': self.load_template('E5'),
            'E6': self.load_template('E6')
        }
    
    def generate_description(self, project_name):
        """Generiert aktualisierte Projektbeschreibung"""
        
        # Repository-Zustand analysieren
        project_state = self.analyze_project_state(project_name)
        
        # Template mit aktuellen Daten fÃ¼llen
        description = self.fill_template(project_name, project_state)
        
        # QualitÃ¤t validieren
        quality_score = self.validate_quality(description)
        
        if quality_score > 80:
            self.save_description(project_name, description)
            self.commit_update(project_name)
            return True
        else:
            print(f"âš ï¸ Quality score for {project_name} below threshold: {quality_score}")
            return False
    
    def analyze_project_state(self, project_name):
        """Analysiert aktuellen Zustand des Projekts im Repository"""
        
        fach = re.match(r'([A-Z]+)', project_name).group(1)
        jahrgangsstufe = re.search(r'(\d+)', project_name).group(1)
        
        state = {
            'fach': fach,
            'jahrgangsstufe': jahrgangsstufe,
            'timestamp': datetime.now().isoformat(),
            'letzte_commits': self.get_recent_commits(project_name),
            'verfÃ¼gbare_materialien': self.scan_materials(fach, jahrgangsstufe),
            'aktuelle_sequenzen': self.find_sequences(fach, jahrgangsstufe),
            'digitale_resources': self.identify_digital_content(fach, jahrgangsstufe)
        }
        
        return state
    
    def get_recent_commits(self, project_name, days=7):
        """Holt recent commits fÃ¼r Projekt-relevante Dateien"""
        
        since = (datetime.now() - timedelta(days=days)).strftime('%Y-%m-%d')
        
        commits = list(self.repo.iter_commits(
            paths=[f"unterricht/*{project_name}*", f"notizen/{project_name[:3]}*"],
            since=since
        ))
        
        return [{
            'message': commit.message.strip(),
            'date': commit.committed_datetime.isoformat(),
            'files': list(commit.stats.files.keys())
        } for commit in commits[:5]]
    
    def scan_materials(self, fach, jahrgangsstufe):
        """Scannt verfÃ¼gbare Unterrichtsmaterialien"""
        
        pattern_paths = [
            f"unterricht/*{fach}{jahrgangsstufe}*",
            f"notizen/{fach}/*{fach}_{jahrgangsstufe}*"
        ]
        
        materials = []
        for pattern in pattern_paths:
            materials.extend(self.repo_path.glob(pattern))
            
        return [{
            'name': mat.name,
            'path': str(mat.relative_to(self.repo_path)),
            'type': 'directory' if mat.is_dir() else 'file',
            'last_modified': datetime.fromtimestamp(mat.stat().st_mtime).isoformat()
        } for mat in materials]
    
    def save_description(self, project_name, description):
        """Speichert generierte Beschreibung"""
        
        output_dir = self.repo_path / "project_descriptions"
        output_dir.mkdir(exist_ok=True)
        
        file_path = output_dir / f"{project_name}_description.md"
        
        with open(file_path, 'w', encoding='utf-8') as f:
            f.write(description)
            
        print(f"âœ… Generated description for {project_name}: {file_path}")
    
    def commit_update(self, project_name):
        """Committet Update zur Versionierung"""
        
        file_path = f"project_descriptions/{project_name}_description.md"
        
        self.repo.index.add([file_path])
        
        commit_message = f"feat({project_name}): auto-update project orientation [{datetime.now().strftime('%Y-%m-%d-%H%M')}]"
        
        self.repo.index.commit(commit_message)
        
        print(f"ğŸ“ Committed update for {project_name}")
